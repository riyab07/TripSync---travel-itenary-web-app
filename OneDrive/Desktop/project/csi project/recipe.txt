class RecipeSuggester:
    def suggest(self, inventory):
        # Dummy rule-based recipes
        available = [item['name'].lower() for item in inventory if float(item['qty']) > 0]

        recipes = []
        if 'tomato' in available and 'onion' in available:
            recipes.append({
                'title': 'Tomato Onion Curry',
                'ingredients': ['tomato', 'onion', 'spices']
            })
        if 'potato' in available and 'peas' in available:
            recipes.append({
                'title': 'Aloo Matar',
                'ingredients': ['potato', 'peas', 'spices']
            })
        if not recipes:
            recipes.append({
                'title': 'No suggestions',
                'ingredients': []
            })
        return recipes
